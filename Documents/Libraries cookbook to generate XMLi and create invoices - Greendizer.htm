Introduction

The libraries can help you construct your XMLi invoices. Using a set of objects that represent the different components of an invoice, you will be able to generate your XMLi string and send it using a POST request through the Greendizer API or by using the send() method in the InvoiceContainer

More details are available on the XMLi reference.

XMLiBuilder

The XMLiBuilder is the object generating the XMLi string. The elements of the invoice will be added to it as we shall see.

  • Python
  • Php
  • Java
from greendizer.xmli import XMLiBuilder
            
builder = XMLiBuilder()
require_once 'Greendizer/XMLi/XMLiBuilder.php';

$builder = new Greendizer_XMLi_XMLiBuilder();
import com.greendizer.api.xmli.XMLiBuilder;

XMLiBuilder builder = new XMLiBuilder();

Invoice

Represents the invoice element of the XMLi invoice.

  • Python
  • Php
  • Java
from greendizer.xmli import *

#Instance of the Invoice object
invoice = Invoice(name="Building and invoice", currency="USD",
                  description="Made with Python", status=INVOICE_DUE,
                  date=date.today(), due_date=date.today() + timedelta(days=75),
                  terms="Specific terms and conditions", custom_id="IJH1234567")
                  
invoice.buyer.name = "Jimi Hendrix"
invoice.buyer.email = "jimi.hendrix@springshield.com"
invoice.buyer.address = Address(street_address="15 rue de Chambéry",
                                city="Paris", zipcode="75015", country="FR")

invoice.shipping.recipient = invoice.buyer

#Add the invoice to the builder
builder.invoices.append(invoice)
require_once 'Greendizer/XMLi/Invoice.php';
require_once 'Greendizer/XMLi/XMLiNamespace.php';
require_once 'Greendizer/XMLi/XMLiBuilder.php';
require_once 'Greendizer/XMLi/Tax.php';
require_once 'Greendizer/XMLi/RateType.php';
require_once 'Greendizer/XMLi/Group.php';
require_once 'Greendizer/XMLi/Line.php';

//Instance of the Invoice object
$invoice = new Greendizer_XMLi_Invoice();
$invoice->setCurrency('EUR');
$invoice->setName('Building an invoice');
$invoice->setDescription('Invoice built with Python');
$invoice->setDate('2011-03-16');
$invoice->setDueDate('2011-12-28');
$invoice->setStatus(Greendizer_XMLi_InvoiceStatus::DUE);

$invoice->getBuyer()->setName("Jimi Hendrix");
$invoice->getBuyer()->setEmail("jimi.hendrix@springshield.com");
$invoice->getBuyer()->setAddress(new Greendizer_XMLi_Address());
$invoice->getBuyer()->getAddress()->setNumber('15');
$invoice->getBuyer()->getAddress()->setStreet('rue de Chambery');
$invoice->getBuyer()->getAddress()->setZipCode('75015');
$invoice->getBuyer()->getAddress()->setCity('Paris');
$invoice->getBuyer()->getAddress()->setCountry('FR');

//Add the invoice to the builder
$builder->addInvoice($invoice);
import com.greendizer.api.xmli.Invoice;

//Instance of the Invoice object
Invoice invoice = new Invoice();
invoice.setCurrency(CurrencyEnum.EUR);

//Setting the buyer\'s address
invoice.getPostalAddress().setNumber("15");
invoice.getPostalAddress().setStreet("rue de Chambéry");
invoice.getPostalAddress().setZipcode("75015");
invoice.getPostalAddress().setCity("Paris");
invoice.getPostalAddress().setCountry("FR");

//Add the invoice to the builder
builder.addChild(invoice);

Group

Represents a group of lines in the invoice

  • Python
  • Php
  • Java
#Initialize a group to organize lines
group = Group(name='Group name', description='Group description')

#Add the group the list of groups in the invoice
invoice.groups.append(group)
//Initialize a group to organize lines
$group = new Greendizer_XMLi_Group();
$group->setName('Group name');
$group->setDescription('Group description');

//Add the group the list of groups in the invoice
$invoice->addGroup($group);
import com.greendizer.api.xmli.Group;

//Initialize a group to organize lines
Group group = new Group();
group.setName("Main name");
group.setDescription("Group description");

//Adding the group to an invoice
invoice.addChild(group);

Tax & Discount

Represents a tax or a discount to be applied to a line.

  • Python
  • Php
  • Java
#Initialize a Tax instance that you'll apply to the lines
tax = Tax(name='Tax name" #VAT for example', rate=19.6,
           description='Description of the tax',
           rate_type=RATE_TYPE_FIXED) #could also be RATE_TYPE_PERCENTAGE

#Initialize work the exact same way
discount = Discount(name="Discount name", description='Discount description',
                    rate=50, rate_type=RATE_TYPE_PERCENTAGE)
//Initialize a Tax instance that you'll apply to the lines
$tax = new Greendizer_XMLi_Tax();
$tax->setName('Tax name'); //VAT for example
$tax->setDescription('Description of the tax');
$tax->setRate(19.6);
$tax->setType(Greendizer_XMLi_RateType::FIXED);		//Could also be PERCENTAGE 

//Initialize work the exact same way
$discount = new Greendizer_XMLi_Discount();
$discount->setName('Discount name');
$discount->setDescription('Discount description');
$discount->setRate(50);
$discount->setType(Greendizer_XMLi_RateType::PERCENTAGE);	//could also be FIXED
import com.greendizer.api.xmli.Group;

//Declaring a Discount

discount.setRate(50);
discount.setType(RateTypeEnum.FIXED); //could also be FIXED

Line

Represents a line in the invoice.

  • Python
  • Php
  • Java

line = Line(name='Service or product name',

            quantity=1, unit_price=5000, date=date.today())
            
line.taxes.append(tax)
line.discounts.append(discount)
group.lines.apppend(line)

group.addChild(line);

Custom Elements

Namespaces

You can declare your custom namespaces as follow

  • Python
  • Php
  • Java
company_ns = "myCompany:http://example.mycompany.com"
company_ns = new XMLNamespace("myCompany", "http://example.mycompany.com");
import com.greendizer.api.xmli.XMLNamespace;

XMLNamespace namespace = new XMLNamespace("myCompany", "http://example.mycompany.com");

Adding elements

You can add custom elements to the body of the invoice, to group, or to line.

  • Python
  • Php
  • Java
group[company_ns]['group_custom_field'] = 'group custom field value'
           
line[company_ns]['line_custom_field'] = "line custom field value"
$group->setCustomElement($company_ns, 'group_custom_field', 'group custom field value');

$invoice->setCustomElement($company_ns, 'line_custom_field', 'line custom field value');
//Adding a custom element to a group
group.setCustomElement(namespace, "group_custom_field", "group custom field value");
